# **토마토**
### 📌 [BOJ G5 7576 토마토](https://www.acmicpc.net/problem/7576)
-------------
### **✔ 풀이 과정 / 학습한 내용**
- 값이 1인 경우만 토마토이므로 해당 위치를 리스트에 추가하고, BFS로 인접한 칸이 0일 경우 날짜를 하루씩 증가시켰다.
그리고 문제의 조건에 따라 처음부터 토마토로 가득찼으면 0 출력, 익지 않은 토마토가 하나라도 있으면 -1 출력, 그외의 경우 날짜를 출력했다.
-------------
### **코드**
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BOJ_7576_토마토 {

	static class Point {
		private int x;
		private int y;
		private int day;
		
		public Point(int x, int y, int day) {
			this.x = x;
			this.y = y;
			this.day = day;
		}

		public int getX() {
			return x;
		}

		public int getY() {
			return y;
		}
		
		public int getDay() {
			return day;
		}
	}
	
	private static int N, M, count, zeroCount;
	private static int[][] grid;
	private static ArrayList<Point> list;
	private static int[] dx = {0, 1, 0, -1};
	private static int[] dy = {-1, 0, 1, 0};
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine(), " ");
		
		M = Integer.parseInt(st.nextToken());
		N = Integer.parseInt(st.nextToken());
		grid = new int[N][M];
		list = new ArrayList<Point>();
		count = 0;
		zeroCount = 0;
		
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			for (int j = 0; j < M; j++) {
				grid[i][j] = Integer.parseInt(st.nextToken());
				// 1인 경우 리스트에 저장하기
				if (grid[i][j] == 1) {
					list.add(new Point(j, i, 0));
				}
				
				// 0인 경우가 있다면 세기
				if (grid[i][j] == 0) {
					zeroCount++;
				}
			}
		}
		
		// 처음부터 0이 없는 가득 저장된 상태면 0 출력하고 종료
		if (zeroCount == 0) {
			System.out.println(0);
			return;
		}
		
		bfs();
	}

	// 너비 우선 탐색
	private static void bfs() {
		Queue<Point> queue = new LinkedList<Point>();
		// 리스트에서 토마토인 좌표 하나씩 꺼내 큐에 추가
		for (Point p : list) {
			queue.offer(p);
		}

		while (!queue.isEmpty()) {
			Point cur = queue.poll();
			int curX = cur.getX();
			int curY = cur.getY();
			int curDay = cur.getDay();
			
			// 큐에 저장된 시간이 현재 기록된 일수보다 크면 업데이트
			if (count < curDay) {
				count = curDay;
			}
			
			for (int d = 0; d < 4; d++) {
				int nextX = curX + dx[d];
				int nextY = curY + dy[d];
				
				// 배열 범위이내가 아니면 PASS
				if (!check(nextX, nextY)) {
					continue;
				}
				
				// 다음 인접한 칸이 0이 아니면 PASS
				if (grid[nextY][nextX] != 0) {
					continue;
				}
				
				// 다음 칸 토마토 익은 상태로 저장
				grid[nextY][nextX] = 1;
				
				// 현재 날짜에서 + 1 하여 큐에 추가
				queue.offer(new Point(nextX, nextY, curDay + 1));
			}
			
		}
		
		getDate();
	}
	
	// 결과값 출력 메서드
	private static void getDate() {	
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				// 익지 않은 토마토가 하나라도 있으면 -1 출력 후 종료
				if (grid[i][j] == 0) {
					System.out.println(-1);
					return;
				}
			}
		}
		
		System.out.println(count);
	}
	
	// 배열 범위 이내인지 체크 메서드
	private static boolean check(int x, int y) {
		if (x < 0 || x >= M || y < 0 || y >= N) {
			return false;
		}
		
		return true;
	}
}
```
